# profile and bashrc
- /private/etc/profile
- /private/etc/bashrc
- /etc/profile # 開機後被 /private/etc/profile 覆蓋
- /etc/bashrc # 開機後被 /private/etc/bashrc 覆蓋
- ~/.profile # 開機不執行 ; 開終端機會執行 (如果檔案 ~/.bash_profile 存在 , 就僅執行 ~/.bash_profile , 不執行 ~/.profile)
- ~/.bashrc # 開機不執行 ; 開終端機不執行
